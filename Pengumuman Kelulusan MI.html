2. Script Validasi NISN (via Google Sheets)

Menggunakan Tabletop.js atau Google Sheets API untuk fetch data:

html
<script src="https://cdn.jsdelivr.net/npm/tabletop@1.6.0/tabletop.min.js"></script>
<script>
const sheetURL = 'https://docs.google.com/spreadsheets/d/1SgUHwPCtzZK_OPjjYjU3kEs0FaSKI1nJEhofRdxrL9A/pubhtml';

let siswaData = [];

window.onload = function() {
  Tabletop.init({
    key: sheetURL,
    simpleSheet: true,
    callback: function(data) {
      siswaData = data;
    }
  });
};

function checkNISN() {
  const inputNISN = document.getElementById('nisnInput').value;
  const siswa = siswaData.find(row => row.NISN === inputNISN);

  if (siswa) {
    localStorage.setItem("siswa", JSON.stringify(siswa));
    window.location.href = "pengumuman.html";
  } else {
    alert("NISN tidak ditemukan!");
  }
}
</script>

3. pengumuman.html - Halaman Kedua (Countdown Page)

html
<script>
const siswa = JSON.parse(localStorage.getItem("siswa"));
document.getElementById("namaSiswa").innerText = siswa.NAMA;

const targetTime = new Date("2025-06-02T17:00:00+07:00").getTime();

function countdown() {
  const now = new Date().getTime();
  const distance = targetTime - now;

  if (distance > 0) {
    const d = Math.floor(distance / (1000 * 60 * 60 * 24));
    const h = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
    const m = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
    const s = Math.floor((distance % (1000 * 60)) / 1000);
    document.getElementById("countdown").innerHTML = `Pengumuman ini dapat dibuka dalam: ${d}d ${h}h ${m}m ${s}s`;
  } else {
    document.getElementById("countdown").innerHTML = 'Waktu Pengumuman Telah Tiba';
    document.getElementById("bukaBtn").style.display = 'block';
  }
}
setInterval(countdown, 1000);
</script>

<p>Selamat datang <span id="namaSiswa"></span> di halaman pengumuman kelulusan.</p>
<p>Silahkan berdoa terlebih dahulu sebelum membuka pengumuman ini.</p>
<div id="countdown" class="text-red-500 font-bold mt-4"></div>
<button id="bukaBtn" style="display:none" onclick="startFinalCountdown()">BUKA</button>

4. Final Countdown Sebelum Tampil Hasil

javascript
function startFinalCountdown() {
  let timer = 10;
  const interval = setInterval(() => {
    document.getElementById("bukaBtn").innerText = `Membuka dalam ${timer} detik...`;
    timer--;
    if (timer < 0) {
      clearInterval(interval);
      window.location.href = "hasil.html";
    }
  }, 1000);
}
